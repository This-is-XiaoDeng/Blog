<!-- wp:heading -->
<h2>进入方法</h2>
<!-- /wp:heading -->

<!-- wp:heading {"level":3} -->
<h3>一、（更快）</h3>
<!-- /wp:heading -->

<!-- wp:list {"ordered":true} -->
<ol><li>进入IP为 10087.s.imc.re 的服务器</li><li>按提示登录/注册</li><li>畅玩</li></ol>
<!-- /wp:list -->

<!-- wp:heading {"level":3} -->
<h3>二、（更稳）</h3>
<!-- /wp:heading -->

<!-- wp:list {"ordered":true} -->
<ol><li>进入IP为 play.imc.re 的服务器</li><li>按提示注册/登录</li><li>右键单击出生点右前方向村民（头上有红色的「分享服务器列表」）</li><li>选择名为「SH10087」的命名牌</li><li>畅玩</li></ol>
